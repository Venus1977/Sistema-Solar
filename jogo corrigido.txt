        // Mover nave
        function moveShip(direction) {
            const spaceship = document.getElementById('spaceship');
            let newX = gameState.shipPosition.x;
            let newY = gameState.shipPosition.y;
            
            const step = 20;
            
            switch(direction) {
                case 'up':
                    newY = Math.max(0, newY - step);
                    break;
                case 'down':
                    newY = Math.min(260, newY + step);
                    break;
                case 'left':
                    newX = Math.max(0, newX - step);
                    break;
                case 'right':
                    newX = Math.min(460, newX + step);
                    break;
            }
            
            gameState.shipPosition.x = newX;
            gameState.shipPosition.y = newY;
            
            spaceship.style.left = newX + 'px';
            spaceship.style.top = newY + 'px';
            
            // Verificar proximidade do alvo
            checkTargetProximity();
        }

        // Verificar proximidade do alvo
        function checkTargetProximity() {
            const target = document.getElementById('target');
            const targetRect = target.getBoundingClientRect();
            const gameRect = document.getElementById('missionGame').getBoundingClientRect();
            
            const targetX = targetRect.left - gameRect.left;
            const targetY = targetRect.top - gameRect.top;
            
            const distance = Math.sqrt(
                Math.pow(gameState.shipPosition.x - targetX, 2) + 
                Math.pow(gameState.shipPosition.y - targetY, 2)
            );
            
            if (distance < 60) {
                document.getElementById('completeBtn').disabled = false;
                target.style.boxShadow = '0 0 20px #4ecdc4';
            } else {
                document.getElementById('completeBtn').disabled = true;
                target.style.boxShadow = 'none';
            }
        }

        function loadQuestion() {
            const question = quizQuestions[gameState.currentQuestion];

        function selectAnswer(selectedIndex) {
            const question = quizQuestions[gameState.currentQuestion];
            const options = document.querySelectorAll('.quiz-option');
            
            // Desativar todas as op√ß√µes
            options.forEach(option => option.disabled = true);
            
            // Marcar resposta correta e incorreta
            options[question.correct].classList.add('correct');
            if (selectedIndex !== question.correct) {
                options[selectedIndex].classList.add('incorrect');
            } else {
                gameState.quizScore++;
                document.getElementById('quizScore').textContent = gameState.quizScore;
            }
            
            // Mostrar feedback
            const feedback = document.getElementById('quizFeedback');
            feedback.innerHTML = `
                <strong>${selectedIndex === question.correct ? '‚úÖ Correto!' : '‚ùå Incorreto!'}</strong><br>
                ${question.explanation}
            `;
            feedback.style.display = 'block';
            
            // Mostrar bot√£o pr√≥xima pergunta ou reiniciar
            if (gameState.currentQuestion < quizQuestions.length - 1) {
                document.getElementById('nextBtn').style.display = 'inline-block';
            } else {
                document.getElementById('restartBtn').style.display = 'inline-block';
                checkAchievements();
            }
        }
